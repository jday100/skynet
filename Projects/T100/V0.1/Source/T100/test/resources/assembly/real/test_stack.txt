MODE 	REAL

DATA
	result 			INTEGER
	a			INTEGER		1
	b			INTEGER		2
	c			INTEGER
	//
	test_add_result		INTEGER
	test_add_a		INTEGER
	test_add_b		INTEGER
	test_add_c			INTEGER
END 

CODE
	MOVE 	SSR	1000
	MOVE 	SPR	0
	//
	PUSH 	result
	PUSH	a
	PUSH	b
	PUSH	c
	CALL 	TEST_ADD
	//
	HALT
END


PROC=TEST_ADD
	POP	test_add_c
	POP	test_add_b
	POP	test_add_a
	POP	test_add_result
	//	
END
