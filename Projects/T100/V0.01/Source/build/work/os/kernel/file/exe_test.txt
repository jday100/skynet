PROC=EXE_LOAD_TEST

    MOVE    AAR                                             @exe_load_test_result

    //
    MOVE    AAR                                             @exe_load_test_result
    CALL    EXE_LOAD_TEST_TABLE

    JZ      EXE_LOAD_TEST_ERROR
    //
    CMT     1
    //DBG
    MOVE    #exe_load_test_result_offset                    1
    RET
EXE_LOAD_TEST_ERROR:
